\begin{topic}{Topology}
  Pair $(M, d)$ is metric space where $M$ is a set and $d: M \times M \rightarrow [0,\infty)$
  is a metric. It satisfies
  \begin{enumerate}
    \item $d(x, z) \leq d(x, y) + d(y, z)$
    \item $d(x, y) = d(y, x)$
    \item $d(x, y) = 0 \Leftrightarrow x = y$
  \end{enumerate}
\end{topic}
