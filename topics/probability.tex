\begin{definition}{Probability Space}
  A measure space $(\Omega, \mathcal{F}, \prob)$, where $\Omega$ is the sample space, $\mathcal{F}$ is the $\sigma$-algebra of events $E$, and $\prob : \mathcal{F} \rightarrow [0, 1]$ is the probability measure.
\end{definition}

\begin{theorem}{Fundamental Identities}
  \begin{enumerate}[leftmargin=*, nosep]
    \item Independence $\prob(E \cap F) = \prob(E)\prob(F)$
    \item Conditional Probability $\prob(E \given F) = \frac{\prob(E \cap F)}{\prob(F)}$
    \item Total Probability $\prob(B) = \sum_i \prob(B \given E_i) \prob(E_i)$
    \item Bayes' Law $\prob(E_i \given B)
      = \frac{ \prob(B \given E_i) \prob(E_i) }{ \sum_j \prob(B \given E_j) \prob(E_j) }$
  \end{enumerate}
\end{theorem}

\begin{theorem}{Linearity of Expectation}
  \[ \E \left[ \sum X_i \right] = \sum \E[X_i] \]
\end{theorem}

\begin{theorem}{Jensen's Inequality}
  If $f$ is convex, then $\E[ f(X) ] \geq f( \E[X] )$.
\end{theorem}
