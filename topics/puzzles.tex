\begin{puzzle}{Permutation Cycles}{Linearity of Expectation}
  A permutation on $[1, n]$ decomposes into disjoint cycles.
  What is the expected number of cycles in a random permutation?
\end{puzzle}

\begin{puzzle}{Lattice Random Walk}{Independent axes, sum diverges}
  Classify simple random walks on $\Z^2$ and $\Z^3$ as
  transient or recurrent.
\end{puzzle}

\begin{puzzle}{k-Suppliers}{Triangle inequality, dominating set}
  Given a set of suppliers $F$ and customers $D$, find a subset $S
  \subseteq F$ with size $|S| \le k$ that minimizes the maximum
  distance from any customer in $D$ to the nearest supplier in $S$.
  \begin{enumerate}
    \item Give a 3-approximation algorithm for this problem.
    \item Prove that there is no better approximation algorithm unless $\P = \NP$.
  \end{enumerate}
\end{puzzle}

\begin{puzzle}{Ant on Rubber Rope}{Harmonic series diverges}
  Ant crawls at $1$ cm/s. Rope stretches by $1$ m/s every second.
  Does the ant reach the end?
\end{puzzle}
